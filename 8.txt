#include <iostream>
#include <string>
using namespace std;

int GetSumRever(int n) 
{
	string str = to_string(n);
	reverse(str.begin(), str.end());
	int r = stoi(str);
	return (r + n);
}

bool Praimoreal(int n) 
{
	n = GetSumRever(n);
	n = GetSumRever(n);
	string str = to_string(n);
	for (int i = 0, j = str.length() - 1; i < str.length() / 2 && j > str.length() / 2; ++i, --j)
		if (str[i] != str[j])
			return false;
	return true;
}

int main() 
{
	int n;
	cin >> n;
	cout << Praimoreal(n);
	return 0;
}
