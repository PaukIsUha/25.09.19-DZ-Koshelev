#include <iostream>
#include <cmath>
using namespace std;

bool Easy(int n) 
{
	if (n == 0)
		return false;
	else if (n == 1 || n == 2)
		return true;
	else
	{
		for (int i = 2; i < sqrt(n); ++i)
			if (n % i == 0)
				return false;
		return true;
	}
}

int PrimeReal(int n) 
{
	int p = 1;
	for (int i = 1; i <= n; ++i)
		if (Easy(i))
			p *= i;
	return p;
}

int main() 
{
	int n;
	cin >> n;
	cout << PrimeReal(n);
	return 0;
}
