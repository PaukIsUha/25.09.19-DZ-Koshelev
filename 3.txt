#include <iostream>
#include <cmath>
using namespace std;

bool Easy(int n) 
{
	for (int i = 2; i <= sqrt(n); ++i)
		if (n % i == 0)
			return false;
	return true;
}

int main() 
{
	int n, count = 4;
	cin >> n;
	for (int i = 0; i < n; ++i) 
	{
		for(;true; ++count)
			if (!Easy(count)) 
			{
				cout << count << " ";
				++count;
				break;
			}
	}
	return 0;
}
