#include <iostream>
using namespace std;

/*
	2020x + 2021y = 78487894
	2020x + 2020y + y = 78487894
	2020(x + y) + y = 78487894
	пусть z = x + y
	2020z + y = 78487894

	{ z = k
	{ y = 78487894 - 2020k
	{ x = 2021k - 78487894

	чтобы корни принадлежали натуральным числам нужно:
	2021k > 78487894
	k > 78487894 / 2021, при этом k принадлежит целым числам
	=> k >= 38837
*/

int main() 
{
	for (long long int k = 38837; k < 38856; ++k) 
		cout << 2021 * k - 78487894 << " " << 78487894 - 2020 * k << endl;
	return 0;
}
